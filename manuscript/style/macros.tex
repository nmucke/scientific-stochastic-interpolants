\NewDocumentCommand{\dist}{o o g}{%
  \ensuremath{
    % Base symbol
    p%
    % Subscript: p_{#1} if first [] is present
    \IfValueT{#1}{_{#1}}%
    % Superscript: p^{#2} if second [] is present
    \IfValueT{#2}{^{#2}}%
    % Variable: (x) if {} is present
    \IfValueT{#3}{(#3)}%
  }
}


\NewDocumentCommand{\distbar}{o o g}{%
  \ensuremath{
    % Base symbol
    \bar{p}%
    % Subscript: p_{#1} if first [] is present
    \IfValueT{#1}{_{#1}}%
    % Superscript: p^{#2} if second [] is present
    \IfValueT{#2}{^{#2}}%
    % Variable: (x) if {} is present
    \IfValueT{#3}{(#3)}%
  }
}

\NewDocumentCommand{\conddist}{o o m m}{%
  \ensuremath{
    p%
    % Subscript part
    \IfValueT{#1}{_{#1}}%
    % Superscript part
    \IfValueT{#2}{^{#2}}%
    % Conditional part with auto-scaling delimiters
    \!\left(#3 \,\middle|\, #4\right)%
  }
}
\NewDocumentCommand{\conddistbar}{o o m m}{%
  \ensuremath{
    \bar{p}%
    % Subscript part
    \IfValueT{#1}{_{#1}}%
    % Superscript part
    \IfValueT{#2}{^{#2}}%
    % Conditional part with auto-scaling delimiters
    \!\left(#3 \,\middle|\, #4\right)%
  }
}

      }%
  }
}
